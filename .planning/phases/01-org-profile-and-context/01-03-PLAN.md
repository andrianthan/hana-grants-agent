```python
import json
import boto3
import psycopg2
from pgvector.psycopg2 import register_vector

def get_connection(secret_arn: str, region: str = "us-west-2"):
    """Get a PostgreSQL connection using credentials from Secrets Manager."""
    sm = boto3.client("secretsmanager", region_name=region)
    secret = json.loads(sm.get_secret_value(SecretId=secret_arn)["SecretString"])
    conn = psycopg2.connect(
        host=secret["host"],
        port=secret.get("port", 5432),
        dbname=secret.get("dbname", "hanna"),
        user=secret["username"],
        password=secret["password"],
        sslmode="require",
    )
    register_vector(conn)
    return conn
```

**scripts/utils/embeddings.py:**
```python
import json
import boto3

BEDROCK = boto3.client("bedrock-runtime", region_name="us-west-2")
MODEL_ID = "amazon.titan-embed-text-v2:0"
DIMENSIONS = 1024

def get_embedding(text: str) -> list[float]:
    """Generate a 1024-dim embedding via Amazon Bedrock Titan Text Embeddings V2."""
    response = BEDROCK.invoke_model(
        modelId=MODEL_ID,
        body=json.dumps({"inputText": text, "dimensions": DIMENSIONS, "normalize": True}),
    )
    return json.loads(response["body"].read())["embedding"]
```

**scripts/utils/chunking.py:**
```python
import re
from typing import List, Dict

# Matches: markdown headers (# ## ###), ALL-CAPS section titles, numbered sections starting with capital
SECTION_PATTERN = re.compile(
    r"^(?:(?:#{1,3}\s+)|(?:[A-Z][A-Z\s&/,:\-]{5,})\s*$|(?:\d+\.\s+[A-Z]))",
    re.MULTILINE,
)

MIN_CHUNK_LENGTH = 50  # Skip trivially small chunks

def chunk_by_section(
    text: str,
    source_file: str,
    doc_type: str,
    funder: str = "",
    year: str = "",
) -> List[Dict]:
    """Split text into chunks at section boundaries. Returns list of chunk dicts."""
    splits = SECTION_PATTERN.split(text)
    headers = SECTION_PATTERN.findall(text)
    chunks = []
    for i, content in enumerate(splits):
        content = content.strip()
        if len(content) < MIN_CHUNK_LENGTH:
            continue
        chunks.append({
            "content": content,
            "section_title": headers[i - 1].strip() if i > 0 and i - 1 < len(headers) else "Introduction",
            "source_file": source_file,
            "doc_type": doc_type,
            "funder": funder,
            "year": year,
        })
    return chunks
```

```python
#!/usr/bin/env python3
"""Initialize the Hanna grants database schema (pgvector, documents, hyde_queries tables)."""
import argparse
import sys
from utils.db import get_connection

SCHEMA_SQL = """
CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE IF NOT EXISTS documents (
    id SERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    embedding vector(1024) NOT NULL,
    source_file TEXT NOT NULL,
    doc_type TEXT NOT NULL,
    section_title TEXT,
    funder TEXT DEFAULT '',
    year TEXT DEFAULT '',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS documents_embedding_idx
    ON documents USING hnsw (embedding vector_cosine_ops)
    WITH (m = 16, ef_construction = 64);

CREATE TABLE IF NOT EXISTS hyde_queries (
    id SERIAL PRIMARY KEY,
    query_text TEXT NOT NULL,
    embedding vector(1024) NOT NULL,
    profile_hash TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);
"""

def main():
    parser = argparse.ArgumentParser(description="Initialize Hanna DB schema")
    parser.add_argument("--secret-arn", required=True, help="Secrets Manager ARN for DB credentials")
    args = parser.parse_args()

    conn = get_connection(args.secret_arn)
    try:
        with conn.cursor() as cur:
            cur.execute(SCHEMA_SQL)
        conn.commit()
        print("Schema initialized: documents table, hyde_queries table, HNSW index created.")
    finally:
        conn.close()

if __name__ == "__main__":
    main()
```

**scripts/extract_pdfs.py:**

CLI script that extracts all PDFs in `org-materials/` subdirectories to text files in `org-materials/.extracted-text/`.

Key behavior:
- Scan 3 directories: `org-materials/grant-applications/`, `org-materials/progress-reports/`, `org-materials/work-plans/`
- For each PDF, generate output filename by replacing spaces with underscores and `.pdf` with `.txt`
- **SKIP extraction if output .txt file already exists** (D-07: reuse FYC_CSS_Proposal.txt, ProbationProposal.txt, SuicidePreventionProposal.txt)
- Use `pdfplumber` to extract text page-by-page, join with newlines
- Print progress: `Extracting: {filename}` / `Skipping (exists): {filename}`
- At end, print summary: `Extracted: {N} new, Skipped: {M} existing, Total: {N+M}`

```python
#!/usr/bin/env python3
"""Extract text from all PDFs in org-materials/ subdirectories."""
import os
import pdfplumber

ORG_DIR = os.path.join(os.path.dirname(__file__), "..", "org-materials")
OUTPUT_DIR = os.path.join(ORG_DIR, ".extracted-text")
SUBDIRS = ["grant-applications", "progress-reports", "work-plans"]

def pdf_to_text(pdf_path: str) -> str:
    pages = []
    with pdfplumber.open(pdf_path) as pdf:
        for page in pdf.pages:
            text = page.extract_text()
            if text:
                pages.append(text)
    return "\n\n".join(pages)

def main():
    os.makedirs(OUTPUT_DIR, exist_ok=True)
    extracted, skipped = 0, 0
    for subdir in SUBDIRS:
        dir_path = os.path.join(ORG_DIR, subdir)
        if not os.path.isdir(dir_path):
            continue
        for fname in sorted(os.listdir(dir_path)):
            if not fname.lower().endswith(".pdf"):
                continue
            out_name = fname.replace(" ", "_").replace(".pdf", ".txt").replace(".PDF", ".txt")
            out_path = os.path.join(OUTPUT_DIR, out_name)
            if os.path.exists(out_path):
                print(f"Skipping (exists): {out_name}")
                skipped += 1
                continue
            pdf_path = os.path.join(dir_path, fname)
            text = pdf_to_text(pdf_path)
            with open(out_path, "w") as f:
                f.write(text)
            print(f"Extracted: {out_name}")
            extracted += 1
    print(f"\nDone. Extracted: {extracted} new, Skipped: {skipped} existing, Total: {extracted + skipped}")

if __name__ == "__main__":
    main()
```

**scripts/ingest_documents.py:**

CLI script that reads all extracted text + 4 supplementary markdown files, chunks them, embeds via Bedrock Titan, and inserts into the `documents` table.

Key behavior:
- Accept `--secret-arn` argument (required)
- **Phase 1: Ingest extracted PDF text files** from `org-materials/.extracted-text/`
  - Infer `doc_type` from filename patterns:
    - Contains "Proposal" or "Application" or "App" -> `application`
    - Contains "Progress_Report" or "Report" -> `progress-report`
    - Contains "Work_Plan" or "Scope_of_Work" -> `work-plan`
    - Default -> `application`
  - Infer `funder` from filename (best-effort: first recognizable funder name, or empty string)
  - Infer `year` from filename (4-digit year pattern, or empty string)
- **Phase 2: Ingest supplementary markdown files** (D-10):
  - `org-materials/FUNDER-DIRECTORY.md` -> doc_type: `reference`
  - `org-materials/METRICS-AND-OUTCOMES.md` -> doc_type: `reference`
  - `org-materials/PARTNER-NETWORK.md` -> doc_type: `reference`
  - `org-materials/WRITING-STYLE-GUIDE.md` -> doc_type: `reference`
- For each file: read text -> `chunk_by_section()` -> for each chunk: `get_embedding(chunk["content"])` -> INSERT into `documents`
- Use batch inserts (one INSERT per chunk with `executemany` or loop)
- Print progress per file and total chunks inserted at end
- Handle the case where `documents` table already has data by using `TRUNCATE documents` with a `--fresh` flag, or simply appending (default)

```python
#!/usr/bin/env python3
"""Ingest org-materials documents into pgvector: chunk, embed, store."""
import argparse
import os
import re
import sys

sys.path.insert(0, os.path.dirname(__file__))
from utils.db import get_connection
from utils.embeddings import get_embedding
from utils.chunking import chunk_by_section

ORG_DIR = os.path.join(os.path.dirname(__file__), "..", "org-materials")
EXTRACTED_DIR = os.path.join(ORG_DIR, ".extracted-text")
SUPPLEMENTARY_FILES = [
    ("FUNDER-DIRECTORY.md", "reference"),
    ("METRICS-AND-OUTCOMES.md", "reference"),
    ("PARTNER-NETWORK.md", "reference"),
    ("WRITING-STYLE-GUIDE.md", "reference"),
]

YEAR_PATTERN = re.compile(r"20\d{2}")

def infer_doc_type(filename: str) -> str:
    lower = filename.lower()
    if any(kw in lower for kw in ["proposal", "application", "app_"]):
        return "application"
    if any(kw in lower for kw in ["progress", "report"]):
        return "progress-report"
    if any(kw in lower for kw in ["work_plan", "scope_of_work"]):
        return "work-plan"
    return "application"

def infer_year(filename: str) -> str:
    match = YEAR_PATTERN.search(filename)
    return match.group() if match else ""

def ingest_file(conn, filepath: str, doc_type: str, funder: str = "", year: str = ""):
    with open(filepath) as f:
        text = f.read()
    source_file = os.path.basename(filepath)
    chunks = chunk_by_section(text, source_file, doc_type, funder, year)
    with conn.cursor() as cur:
        for chunk in chunks:
            embedding = get_embedding(chunk["content"])
            cur.execute(
                "INSERT INTO documents (content, embedding, source_file, doc_type, section_title, funder, year) VALUES (%s, %s, %s, %s, %s, %s, %s)",
                (chunk["content"], embedding, chunk["source_file"], chunk["doc_type"],
                 chunk["section_title"], chunk["funder"], chunk["year"]),
            )
    conn.commit()
    return len(chunks)

def main():
    parser = argparse.ArgumentParser(description="Ingest org documents into pgvector")
    parser.add_argument("--secret-arn", required=True, help="Secrets Manager ARN for DB credentials")
    parser.add_argument("--fresh", action="store_true", help="Truncate documents table before ingesting")
    args = parser.parse_args()

    conn = get_connection(args.secret_arn)
    total_chunks = 0

    if args.fresh:
        with conn.cursor() as cur:
            cur.execute("TRUNCATE documents RESTART IDENTITY")
        conn.commit()
        print("Truncated documents table.")

    # Phase 1: Extracted PDF text files
    if os.path.isdir(EXTRACTED_DIR):
        for fname in sorted(os.listdir(EXTRACTED_DIR)):
            if not fname.endswith(".txt"):
                continue
            filepath = os.path.join(EXTRACTED_DIR, fname)
            doc_type = infer_doc_type(fname)
            year = infer_year(fname)
            n = ingest_file(conn, filepath, doc_type, year=year)
            total_chunks += n
            print(f"Ingested: {fname} -> {n} chunks (type={doc_type})")

    # Phase 2: Supplementary markdown files
    for md_file, doc_type in SUPPLEMENTARY_FILES:
        filepath = os.path.join(ORG_DIR, md_file)
        if not os.path.exists(filepath):
            print(f"Warning: {md_file} not found, skipping")
            continue
        n = ingest_file(conn, filepath, doc_type)
        total_chunks += n
        print(f"Ingested: {md_file} -> {n} chunks (type={doc_type})")

    conn.close()
    print(f"\nDone. Total chunks ingested: {total_chunks}")

if __name__ == "__main__":
    main()
```
