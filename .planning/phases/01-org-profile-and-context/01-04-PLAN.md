```python
#!/usr/bin/env python3
"""Generate a HyDE (Hypothetical Document Embedding) query from Hanna's org profile.

This is a one-time CLI script run after DB init and document ingestion.
The HyDE query is a hypothetical ideal grant description generated by GPT-4o
that captures Hanna's mission, programs, geography, and populations served.
The Prospector agent (Phase 3) uses the embedded HyDE query for semantic
similarity search against incoming grant opportunities.

Auto-regeneration: Stores a SHA-256 hash of ORG-PROFILE.md alongside the query.
On each pipeline run, the hash is compared — if the profile changed, HyDE regenerates.
"""
import argparse
import hashlib
import os
import sys

from openai import OpenAI

sys.path.insert(0, os.path.dirname(__file__))
from utils.db import get_connection
from utils.embeddings import get_embedding

PROFILE_PATH = os.path.join(os.path.dirname(__file__), "..", "org-materials", "ORG-PROFILE.md")

HYDE_SYSTEM_PROMPT = "You are an expert grant writer and nonprofit funding specialist."

HYDE_USER_PROMPT = """You are a grant research specialist. Given the following nonprofit
organization profile, write a hypothetical ideal grant opportunity description
that would be a perfect match for this organization. The description should read
like a real grant announcement and include:

- Grant title
- Funder name and type
- Purpose and focus area
- Eligible applicants
- Award range
- Key requirements
- Geographic focus

The description should capture the organization's mission, programs, geography,
and population served. Write it as if it were a real RFP/RFA announcement.

Organization Profile:
{profile_text}

Write the hypothetical ideal grant description:"""


def compute_profile_hash(profile_path: str) -> str:
    """Compute SHA-256 hash of the org profile file."""
    with open(profile_path) as f:
        content = f.read()
    return hashlib.sha256(content.encode("utf-8")).hexdigest()


def generate_hyde_query(profile_path: str, openai_api_key: str) -> tuple[str, str]:
    """Generate HyDE query text and profile hash.

    Returns: (query_text, profile_hash)
    """
    with open(profile_path) as f:
        profile_text = f.read()

    profile_hash = hashlib.sha256(profile_text.encode("utf-8")).hexdigest()

    client = OpenAI(api_key=openai_api_key)
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": HYDE_SYSTEM_PROMPT},
            {"role": "user", "content": HYDE_USER_PROMPT.format(profile_text=profile_text)},
        ],
        temperature=0.7,
        max_tokens=1500,
    )

    query_text = response.choices[0].message.content
    return query_text, profile_hash


def check_existing_hash(conn, profile_hash: str) -> bool:
    """Check if a HyDE query with this profile hash already exists."""
    with conn.cursor() as cur:
        cur.execute("SELECT COUNT(*) FROM hyde_queries WHERE profile_hash = %s", (profile_hash,))
        count = cur.fetchone()[0]
    return count > 0


def store_hyde_query(conn, query_text: str, embedding: list[float], profile_hash: str):
    """Insert HyDE query into the hyde_queries table."""
    with conn.cursor() as cur:
        cur.execute(
            "INSERT INTO hyde_queries (query_text, embedding, profile_hash) VALUES (%s, %s, %s)",
            (query_text, embedding, profile_hash),
        )
    conn.commit()


def main():
    parser = argparse.ArgumentParser(description="Generate HyDE query from Hanna org profile")
    parser.add_argument("--secret-arn", required=True, help="Secrets Manager ARN for DB credentials")
    parser.add_argument("--openai-api-key", required=False, help="OpenAI API key (or set OPENAI_API_KEY env var)")
    parser.add_argument("--force", action="store_true", help="Regenerate even if profile hash matches existing")
    parser.add_argument("--profile", default=PROFILE_PATH, help="Path to ORG-PROFILE.md")
    args = parser.parse_args()

    openai_api_key = args.openai_api_key or os.environ.get("OPENAI_API_KEY")
    if not openai_api_key:
        print("Error: OpenAI API key required. Set OPENAI_API_KEY or pass --openai-api-key")
        sys.exit(1)

    conn = get_connection(args.secret_arn)

    # Check if regeneration is needed
    profile_hash = compute_profile_hash(args.profile)
    if not args.force and check_existing_hash(conn, profile_hash):
        print(f"HyDE query already exists for current profile hash ({profile_hash[:12]}...). Use --force to regenerate.")
        conn.close()
        return

    print(f"Profile hash: {profile_hash[:12]}...")
    print("Generating HyDE query via GPT-4o...")

    query_text, _ = generate_hyde_query(args.profile, openai_api_key)
    print(f"Generated HyDE query ({len(query_text)} chars)")

    print("Embedding HyDE query via Bedrock Titan (1024 dims)...")
    embedding = get_embedding(query_text)

    print("Storing in hyde_queries table...")
    store_hyde_query(conn, query_text, embedding, profile_hash)

    conn.close()
    print("Done. HyDE query stored with profile hash for auto-regeneration.")
    print(f"\nHyDE query preview:\n{query_text[:500]}...")


if __name__ == "__main__":
    main()
```
